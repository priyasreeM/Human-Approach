import java.util.*;
class Rhombus
{
	public static void main(String arg[])
	{
		int n=5;
		for(int i=1;i<=n;i++)
		{
			for(int k=2;k<=i;k++)
			{
				System.out.print(" ");
			}
			for(int j=1;j<=n;j++)
			{
				System.out.print("*");
			}
		System.out.println();		
		}
	}
}
