import java.util.*;
class HalfDiamondPatternPrg
{
	public static void main(String arg[])
	{
		int n=5;
		for(int i=1;i<=n*2-1;i++)
		{
			if(i<=n)
			{
				for(int j=1;j<=i;j++)
				{
					System.out.print("*");
				}
			System.out.println();
			}
			else
			{
				for(int k=1;k<=(n*2)-i;k++)
				{
					System.out.print("*");
				}
				System.out.println();
			}
		}
	}
}
